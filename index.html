export default { async fetch(request) { // Configurazione Header CORS per permettere al sito di comunicare con il worker const corsHeaders = { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Methods': 'GET, POST, OPTIONS', 'Access-Control-Allow-Headers': 'Content-Type, Authorization', }; // Risposta immediata per le richieste di controllo (Pre-flight) if (request.method === 'OPTIONS') { return new Response(null, { headers: corsHeaders }); } const url = new URL(request.url); // Costruiamo l'URL di destinazione verso MiniMax // Se chiami il worker senza path, di default va su video_generation let apiPath = url.pathname; if (apiPath === '/' || apiPath === '/proxy') { apiPath = '/v1/video_generation'; } const targetUrl = 'https://api.minimax.io' + apiPath + url.search; try { const bodyText = (request.method === 'POST') ? await request.text() : null; const response = await fetch(targetUrl, { method: request.method, headers: { 'Content-Type': 'application/json', 'Authorization': request.headers.get('Authorization') || '', }, body: bodyText }); const data = await response.text(); return new Response(data, { status: response.status, headers: { ...corsHeaders, 'Content-Type': 'application/json', }, }); } catch (err) { return new Response(JSON.stringify({ error: "Errore del Proxy: " + err.message }), { status: 500, headers: corsHeaders }); } } };
